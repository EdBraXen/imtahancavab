#1 Qüvvə

Təsvir

İki müxtəlif k və n ədədləri verilir. Yalnız bit əməliyyatlarından istifadə etməklə 2^k + 2^n ifədəsinin qiymətini verin. Giriş İki müxtəlif k və n (0 ≤ k, n ≤30) ədədləri. Çıxış 2^k + 2^n ifadəsinin qiymətini verin. Giriş #1 0 1 Cavab #1 3

Həll Kodu (C++)

solution.cpp 
long long calculatePowerSum(int k, int n) {
    // 1LL (long long) istifadə edirik ki, 31-ci bitdə daşma olmasın.
    // 1LL << k  riyazi olaraq 2^k-ya bərabərdir.
    long long val1 = (1LL << k);
    long long val2 = (1LL << n);
    
    // k və n müxtəlif olduğu üçün bitlər fərqli mövqelərdədir.
    // Bu halda bitwise OR (|) toplama (+) ilə eyni nəticəni verir.
    return val1 | val2;
}

Həll Kodu (C++)

solution.cpp 
long long correctSolution(long long a, long long b, long long n) {
    // Variant 1: n * a (hamısı tək-tək)
    // Variant 2: (n / 3) * b + (n % 3) * a (paketlər + qalan təklər)
    return min(n * a, (n / 3) * b + (n % 3) * a);
}

Vacib qeyd: Yoxlama zamanı boşluqları (" ") nəzərə almayın. Yəni "ey baki fika ye" mətni boşluqlar silindikdən sonra "eybakifikaye" olur və palindrome sayılır. Həll Kodu (C++)

solution.cpp 
bool correctSolution(char text[], int length) {
    string cleaned = "";
    for (int i = 0; i < length; i++) {
        if (text[i] != ' ') {
            cleaned += text[i];
        }
    }
    
    int n = cleaned.length();
    for (int i = 0; i < n / 2; i++) {
        if (cleaned[i] != cleaned[n - 1 - i]) {
            return false;
        }
    }
    return true;
}

Həll Kodu (C++)

solution.cpp 
int correctSolution(char expr[], int length) {
    int count = 0;
    for (int i = 0; i < length; i++) {
        if (expr[i] == '+' || expr[i] == '-' || expr[i] == '%') {
            count++;
        } else if (expr[i] == '*') {
            if (i + 1 < length && expr[i+1] == '*') {
                i++; // İkinci ulduzu keçirik
            }
            count++;
        } else if (expr[i] == '/') {
            if (i + 1 < length && expr[i+1] == '/') {
                i++; // İkinci slash-ı keçirik
            }
            count++;
        }
    }
    return count;
}

void correctSolution(string s, int &textCount, int &digitCount, int &mixedCount) {
    bool hasLetter = false, hasDigit = false, hasOther = false;
    
    for (char c : s) {
        if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) hasLetter = true;
        else if (c >= '0' && c <= '9') hasDigit = true;
        else hasOther = true;
    }
    
    if (hasLetter && !hasDigit && !hasOther) textCount++;
    else if (!hasLetter && hasDigit && !hasOther) digitCount++;
    else mixedCount++;
}

    int left = 0, right = n - 1;

    }
}

Parametrlər: string s: açıq mətn int k: hərflərin sürüşdürülmə sayı Qaytarır: string: şifrələnmiş mətn Giriş Formatı İlk sətirdə n: s uzunluğu İkinci sətirdə s: açıq mətn Üçüncü sətirdə k: sürüşdürmə sayı Məhdudiyyətlər 1 ≤ n ≤ 100 0 ≤ k ≤ 100 s ASCII simvollardan ibarət, boşluqsuz

string correctCaesarCipher(string s, int k) {
    string result = "";
    int shift = k % 26; // Sürüşdürməni 26-ya modul edirik

    for (char c : s) {
        if (c >= 'a' && c <= 'z') {
            // Kiçik hərflər üçün
            char base = 'a';
            char encrypted_char = (char)(((c - base + shift) % 26) + base);
            result += encrypted_char;
        } else if (c >= 'A' && c <= 'Z') {
            // Böyük hərflər üçün
            char base = 'A';
            char encrypted_char = (char)(((c - base + shift) % 26) + base);
            result += encrypted_char;
        } else {
            // Simvollar şifrələnmir
            result += c;
        }
    }
    return result;
}

İkilik sistemində ədəd yalnız 0 və 1 rəqəmlərindən ibarətdir. Bir ədədin ikilik sistemdəki təmsili, ardıcıl olaraq 2-yə bölünərək və hər bir bölmə nəticəsində qalanları toplamaqla tapılır. Tapşırıq: Verilmiş bir tam ədədin ikilik sistemdəki təmsilini qaytaran decToBin funksiyasını yazın. Funksiya Təsviri: int num — onluq sistemdə ədəd. Qaytarır: String — ədədin ikilik (binary) sistemdəki təmsili. Giriş Formatı: Bir tam ədəd num (-10⁶ ≤ num ≤ 10⁶). Məhdudiyyətlər: num yalnız tam ədədlərdən ibarət olacaq.

Həll Kodu (C++)

solution.cpp 
string correctDecToBin(int num) {
    if (num == 0) return "0";
    
    bool isNegative = num < 0;
    // Modul qiyməti üzərində işləyirik
    int absNum = abs(num); 
    
    string binary = "";
    
    while (absNum > 0) {
        // Qalıq (0 və ya 1)
        int remainder = absNum % 2; 
        // Qalıq nəticənin əvvəlinə əlavə olunur (avtomatik tərs sıraya görə)
        binary = to_string(remainder) + binary; 
        
        absNum /= 2;
    }
    
    // Mənfi işarəsini əlavə edirik
    if (isNegative) {
        return "-" + binary;
    }
    
    return binary;
}

string correctSolution(int num) {
    if (num == 0) return "sıfır";

string correctDecToOctal(int num) {

Massivin Tarazlıq Nöqtəsi (Equilibrium Point) Massivin tarazlıq nöqtəsi elə bir indeksdir ki, həmin indeksdən solda qalan elementlərin cəmi, həmin indeksdən sağda qalan elementlərin cəminə bərabər olsun. Şərt: Sizə tam ədədlərdən ibarət bir massiv verilir. Massivin ilk tarazlıq nöqtəsinin indeksini tapın. Əgər belə bir nöqtə yoxdursa, -1 qaytarın. Nümunələr: [1, 3, 5, 2, 2] -> İndeks: 2 (Çünki 1+3 = 4 və 2+2 = 4. İndeks 2-dəki '5' tarazlıq nöqtəsidir). [1, 2, 3] -> -1 (Tarazlıq nöqtəsi yoxdur). [10, -10, 5] -> İndeks: 2 (Sol cəm 10 + (-10) = 0, sağ cəm isə boş olduğu üçün 0 hesab edilir).

Həll Kodu (C++)

solution.cpp 
int correctSolution(int arr[], int n) {
    long long totalSum = 0;
    long long leftSum = 0;

    // Birinci keçid: Ümumi cəm
    for (int i = 0; i < n; i++) {
        totalSum += arr[i];
    }

    // İkinci keçid: Tarazlıq yoxlanışı
    for (int i = 0; i < n; i++) {
        // Sağ cəmi hesabla və yoxla
        if (leftSum == (totalSum - leftSum - arr[i])) {
            return i;
        }
        leftSum += arr[i];
    }

    return -1;
}

Sizə tam ədədlərdən ibarət height massivi verilir. Bu massivdəki hər bir ədəd eni 1 vahid olan sütunların hündürlüyünü təmsil edir. Yağış yağdıqdan sonra bu sütunların arasında nə qədər suyun toplana biləcəyini hesablayın. Nümunə: [0, 1, 0, 2, 1, 0, 1, 3, 2, 1, 2, 1] -> Cəmi: 6 vahid su.

Həll Kodu (C++)

solution.cpp 
int correctSolution(int height[], int n) {
    if (n <= 2) return 0; // 2 və ya daha az sütun varsa, su yığılmaz

    int left = 0;
    int right = n - 1;
    
    int leftMax = 0;
    int rightMax = 0;
    
    int water = 0;

    while (left < right) {
        if (height[left] < height[right]) {
            if (height[left] >= leftMax) {
                leftMax = height[left];
            } else {
                water += leftMax - height[left];
            }
            left++;
        } else {
            if (height[right] >= rightMax) {
                rightMax = height[right];
            } else {
                water += rightMax - height[right];
            }
            right--;
        }
    }
    return water;
}

Nümunə: [-2, 1, -3, 4, -1, 2, 1, -5, 4] -> Cəm: 6 İzah: [4, -1, 2, 1] yarım-massivinin cəmi 6-dır və bu, ən böyükdür.

Həll Kodu (C++)

solution.cpp 
int correctSolution(int arr[], int n) {
    // İlk elementi başlanğıc maksimum kimi qəbul edirik
    int maxSoFar = arr[0];
    int currentMax = arr[0];

    for (int i = 1; i < n; i++) {
        // Ya cari ədədi tək götürürük, ya da əvvəlki cəmə əlavə edirik
        // (Hansı daha böyükdürsə)
        currentMax = std::max(arr[i], currentMax + arr[i]);
        
        // Ümumi maksimumu yeniləyirik
        maxSoFar = std::max(maxSoFar, currentMax);
    }
    return maxSoFar;
}

solution.cpp 
class Terminal {
public:
    int balance;
    bool isBlocked;
    int wrongAttempts;
    bool isVIP;

    Terminal() {
        balance = 0;
        isBlocked = false;
        wrongAttempts = 0;
        isVIP = false;
    }

    void enterPin(int code) {
        // Bloklanıbsa heç pin kod da yoxlamır
        if (isBlocked) {
            return;
        }

        if (code == 1234) {
            wrongAttempts = 0; // Uğurlu giriş
        } else {
            wrongAttempts++;
            if (wrongAttempts >= 3) {
                isBlocked = true; // 3 səhv -> Blok
            }
        }
    }

    void deposit(int amount) {
        if (isBlocked) return; // Blok yoxlanışı
        balance += amount;
    }

    void makePayment(int amount) {
        if (isBlocked) return; // Blok yoxlanışı

        // Komissiya hesabı
        int fee = 2;
        if (isVIP) {
            fee = 0;
        }

        int totalNeeded = amount + fee;

        // Pul çatırsa ödəniş et
        if (balance >= totalNeeded) {
            balance -= totalNeeded;
        }
    }

    void buyVip() {
        if (isBlocked) return; // Blok yoxlanışı
        
        // Artıq VIP deyilsə və pulu çatırsa
        if (!isVIP && balance >= 100) {
            balance -= 100;
            isVIP = true;
        }
    }

    void resetBlock() {
        // Bu metod blokda olsa belə işləyir (Texniki servis)
        isBlocked = false;
        wrongAttempts = 0;
    }
};

class Airplane {
public:
    int fuel;
    int altitude;
    int engineHeat;
    bool isCrashed;

    Airplane() {
        fuel = 100;
        altitude = 0;
        engineHeat = 0;
        isCrashed = false;
    }

    void takeOff() {
        if (isCrashed) return;
        // Yerdədirsə və yanacaq varsa qalx
        if (altitude == 0 && fuel > 0) {
            altitude = 1000;
        }
    }

    void flyFast() {
        if (isCrashed || altitude == 0) return;

        fuel -= 20;
        engineHeat += 30;

        // Ardıcıl yoxlama: Əvvəl yanacaq, sonra istilik
        if (fuel < 0) {
            isCrashed = true;
        } else if (engineHeat > 100) {
            isCrashed = true;
        }
    }

    void flySlow() {
        if (isCrashed || altitude == 0) return;

        fuel -= 10;
        engineHeat += 10;

        if (fuel < 0) {
            isCrashed = true;
        } else if (engineHeat > 100) {
            isCrashed = true;
        }
    }

    void coolDown() {
        if (isCrashed) return;
        engineHeat -= 20;
        if (engineHeat < 0) {
            engineHeat = 0;
        }
    }

    void land() {
        if (isCrashed) return;
        altitude = 0;
        engineHeat = 0; // Yerə enəndə mühərrik soyuyur
    }
};

class Barista {
public:
    int beans;
    int milk;
    int cups;
    int money;
    int dirt;
    int reputation;

    Barista() {
        beans = 50;
        milk = 100;
        cups = 5;
        money = 0;
        dirt = 0;
        reputation = 50;
    }

    void makeEspresso() {
        // 1. Maşın çox çirklidirsə, heç nə olmur
        if (dirt > 80) return;

        // 2. Resursları yoxlayırıq
        if (beans >= 10 && cups >= 1) {
            // Uğurlu
            beans -= 10;
            cups -= 1;
            money += 30;
            dirt += 10;
            reputation += 2;
        } else {
            // Uğursız (Resurs yoxdur)
            reputation -= 5;
        }
        
        // Limitlər
        if (dirt > 100) dirt = 100;
        if (reputation > 100) reputation = 100;
        if (reputation < 0) reputation = 0;
    }

    void makeLatte() {
        if (dirt > 80) return;

        if (beans >= 10 && milk >= 20 && cups >= 1) {
            beans -= 10;
            milk -= 20;
            cups -= 1;
            money += 50;
            dirt += 15;
            reputation += 3;
        } else {
            reputation -= 5;
        }

        if (dirt > 100) dirt = 100;
        if (reputation > 100) reputation = 100;
        if (reputation < 0) reputation = 0;
    }

    void cleanMachine() {
        dirt = 0;
    }

    void buySupplies() {
        if (money >= 100) {
            money -= 100;
            beans += 50;
            milk += 100;
            cups += 10;
        }
    }
};

Günün sonunda isə ən kritik məqam gəlir – ziyafət vaxtı (feedPopulation); sənin şəhərindəki hər bir canlı (həm işçi, həm əsgər) 2 pay yemək yeyir; əgər anbarında kifayət qədər yemək varsa, hamı doyur, yox əgər yemək bitərsə, dəhşətli aclıq başlayır və çatışmayan hər iki pay yemək üçün bir insan ölür, sağ qalanların kədəri isə xoşbəxtliyi 20 bal aşağı salır. Xalqın gözündə yenidən ucalmaq üçün böyük bir festival keçirə bilərsən (holdFestival); bunun üçün 50 ərzaq və 30 qızıl qurban versən, insanların xoşbəxtliyi 20 bal artar və şəhərdə üsyan varsa, dərhal sakitləşər. Amma çox ehtiyatlı ol (checkStatus); əgər xalqın xoşbəxtliyi 20-dən aşağı düşərsə, onlar sənə qarşı üsyan qaldıracaq (isRevolting), üsyan edən xalq nə işləyər, nə vergi verər; və ən pisi odur ki, əgər üsyan başlayan anda səni qoruyacaq heç bir əsgərin yoxdursa, qəzəbli kütlə sarayına hücum edib şəhəri darmadağın edəcək (isDestroyed), ya da aclıqdan hamı ölərsə, xarabalıqların kralı olaraq qalacaqsan.

solution.cpp 
class MedievalCity {
public:
    int population;
    int food;
    int wood;
    int gold;
    int soldiers;
    int happiness;
    bool isRevolting;
    bool isDestroyed;

    MedievalCity() {
        population = 10;
        food = 100;
        wood = 50;
        gold = 50;
        soldiers = 0;
        happiness = 80;
        isRevolting = false;
        isDestroyed = false;
    }

    void assignWorkers(int farmers, int lumberjacks) {
        if (isDestroyed || isRevolting) return;

        // İşçi sayı əhalidən çox ola bilməz
        if (farmers + lumberjacks > population) return;

        // İstehsal
        food += farmers * 5;
        wood += lumberjacks * 3;

        // Boş qalanlar vergi verir
        int idlePeople = population - (farmers + lumberjacks);
        gold += idlePeople * 1;

        // İşləmək xoşbəxtliyi azaldır
        happiness -= 2;
    }

    void recruitSoldier() {
        if (isDestroyed || isRevolting) return;

        if (gold >= 20 && wood >= 10 && food >= 10 && population >= 1) {
            gold -= 20;
            wood -= 10;
            food -= 10;
            population -= 1;
            soldiers += 1;
            happiness += 5;
        }
    }

    void feedPopulation() {
        if (isDestroyed) return;

        int totalPeople = population + soldiers;
        int neededFood = totalPeople * 2;

        if (food >= neededFood) {
            food -= neededFood;
        } else {
            // Aclıq rejimi
            // Qalan yemək hamıya çatmır
            // Neçə nəfər ac qalır? 
            // Qayda: Çatışmayan hər 2 yemək üçün 1 nəfər ölür.
            // Bu o deməkdir ki: food / 2 qədər insan sağ qalır.
            // Qalanlar (total - sağ qalan) ölür.
            
            int survivors = food / 2;
            int deaths = totalPeople - survivors;
            
            food = 0; // Yemək bitdi
            happiness -= 20; // İnsanlar bədbəxtdir

            // Ölümləri çıxmaq (Əvvəlcə mülki əhali, sonra əsgərlər)
            if (population >= deaths) {
                population -= deaths;
            } else {
                // Əhali bitdi, qalan ölümlər əsgərlərdən çıxılır
                deaths -= population;
                population = 0;
                soldiers -= deaths;
                if (soldiers < 0) soldiers = 0;
            }

            if (population + soldiers == 0) {
                isDestroyed = true;
            }
        }
    }

    void holdFestival() {
        if (isDestroyed) return;

        if (food >= 50 && gold >= 30) {
            food -= 50;
            gold -= 30;
            happiness += 20;
            // Festival üsyanı yatırır
            isRevolting = false;
        }
    }

    void checkStatus() {
        // Limitlər
        if (happiness > 100) happiness = 100;
        if (happiness < 0) happiness = 0;

        // Üsyan şərti
        if (happiness < 20) {
            isRevolting = true;
        }

        // Şəhərin dağılması (Üsyan var və əsgər yoxdursa)
        if (isRevolting && soldiers == 0) {
            isDestroyed = true;
        }
    }
};

class CastleSiege {

        if (arrows >= 10) {
            arrows -= 10;

            if (enemies < 0) {
                enemies = 0;
            }

        }

            enemies -= 25;

                enemies = 0;
            }
            // Bu metodda düşmən divara vura bilmir
        }
    }

        }
    }
